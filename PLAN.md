## 3. Backend — `server.js`

Minimal Express app:

```js
import express from "express";
import Database from "better-sqlite3";
import fs from "node:fs";

const DATA_DIR = "/data";
const DB_PATH = `${DATA_DIR}/scores.db`;
fs.mkdirSync(DATA_DIR, { recursive: true });

const db = new Database(DB_PATH);
db.exec(`
  CREATE TABLE IF NOT EXISTS scores (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL,
    score       INTEGER NOT NULL,
    level       INTEGER NOT NULL,
    seconds     INTEGER NOT NULL,
    kg_lost     INTEGER NOT NULL,
    created_at  INTEGER NOT NULL
  );
  CREATE INDEX IF NOT EXISTS idx_scores_score_desc ON scores(score DESC);
`);

const app = express();
app.use(express.json({ limit: "4kb" }));
app.use(express.static("public"));

app.get("/api/scores", (req, res) => {
  const rows = db.prepare(`
    SELECT id, name, score, level, seconds, kg_lost
    FROM scores
    ORDER BY score DESC, id ASC
    LIMIT 20
  `).all();
  res.json(rows);
});

app.post("/api/scores", (req, res) => {
  const { name, score, level, seconds, kg_lost } = req.body || {};
  // light validation
  if (typeof name !== "string" || name.trim().length < 1 || name.length > 30) {
    return res.status(400).json({ error: "invalid name" });
  }
  if (![score, level, seconds, kg_lost].every(n => Number.isFinite(n) && n >= 0)) {
    return res.status(400).json({ error: "invalid numbers" });
  }
  const info = db.prepare(`
    INSERT INTO scores (name, score, level, seconds, kg_lost, created_at)
    VALUES (?, ?, ?, ?, ?, ?)
  `).run(name.trim().slice(0, 30), score|0, level|0, seconds|0, kg_lost|0, Date.now());
  res.status(201).json({ id: info.lastInsertRowid });
});

app.listen(3000, () => console.log("Thozempic on :3000"));
```

## 4. Frontend changes (in `public/index.html`)

### What goes away
- `localStorage.thozempic_scores` read/write
- `loadLeaderboard()` and `saveLeaderboardWithCurrent()` functions

### What replaces it

```js
async function submitScore(summary) {
  try {
    const res = await fetch("/api/scores", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        name: summary.name,
        score: summary.total,
        level: summary.levelReached,
        seconds: summary.seconds,
        kg_lost: summary.kgLost
      })
    });
    const data = await res.json();
    return data.id; // server-assigned row id
  } catch (e) {
    console.error("score submit failed", e);
    return null;
  }
}

async function fetchLeaderboard() {
  try {
    const res = await fetch("/api/scores");
    return await res.json();
  } catch (e) {
    console.error("leaderboard fetch failed", e);
    return null;
  }
}
```

### Flow on game over

```
finalizeAndGoToGameOver(won)
  → compute breakdown (same as today)
  → goToGameOver()         (scene flips immediately, breakdown rendered)
  → async submitScore() then fetchLeaderboard()
  → store result in lastRunSummary.leaderboard + .currentEntryId
  → next render shows the rows
```

While the network calls are in flight, `drawGameOverScreen()` shows
"Loading scores…" in the leaderboard area. On failure it shows
"Leaderboard unavailable" but the breakdown + Play Again button stay
functional.

Current player's row is highlighted by **matching id from POST response**
to id in GET response.

## 7. Limb position fix

Current anchors in `drawThomasBody`:

```js
const hipY      = -BODY_LEG_H;                        // -99
const shoulderY = hipY - BODY_TORSO_H;                // -191
const headTopY  = shoulderY - BODY_HEAD_GAP - BODY_HEAD_H;
```

Issue (from your screenshot):
- arms sit too high and too wide → look detached at the shoulder
- legs sit too low → visible gap between torso bottom and leg tops

Fix using per-limb attach offsets, leaving torso/head anchors alone:

```js
const ARM_ATTACH_DY  = Math.round(BODY_TORSO_H * 0.15);  // 14 down
const LEG_ATTACH_DY  = Math.round(BODY_TORSO_H * 0.20);  // 18 up
const ARM_ATTACH_DX  = Math.round(100 * 0.10);           // 10 inward

const armPivotY = shoulderY + ARM_ATTACH_DY;             // arms hang lower
const legPivotY = hipY      - LEG_ATTACH_DY;             // legs start higher

// existing per-side spread, tightened:
const armPivotL = -(BODY_SHOULDER_DX - ARM_ATTACH_DX);   // -24
const armPivotR =   BODY_SHOULDER_DX - ARM_ATTACH_DX;    //  24
```

Then in the existing `drawLimb()` calls, replace `(±BODY_SHOULDER_DX,
shoulderY)` with `(armPivotL/R, armPivotY)`, and `(±BODY_HIP_DX, hipY)`
with `(±BODY_HIP_DX, legPivotY)`.

Sprite sizes, walk swing, render order, head position — **unchanged**.

I'll preview after wiring and bump deltas ±5px if it still looks off.
